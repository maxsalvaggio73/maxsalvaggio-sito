Guida SQL e Regole di Sicurezza

Esegui questa query in Supabase per configurare l'anagrafica estesa dei clienti e le regole di Storage:

-- 1. Aggiunge le colonne anagrafiche complete alla tabella profiles alter table public.profiles add column if not exists is_admin boolean default false, add column if not exists file_order jsonb default '[]'::jsonb, add column if not exists full_name text, add column if not exists nationality text, add column if not exists codice_fiscale text, add column if not exists dob date, add column if not exists pob text, add column if not exists birth_province text, add column if not exists birth_country text, add column if not exists address text, add column if not exists residence_city text, add column if not exists residence_province text, add column if not exists residence_country text, add column if not exists phone_number text, add column if not exists social_instagram text, add column if not exists social_facebook text, add column if not exists social_other text, add column if not exists onboarding_completed boolean default false, add column if not exists documents_approved boolean default false; -- 2. Permette a te (Amministratore) di gestire tutto lo storage create policy "Admins can select files" on storage.objects for select using ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); create policy "Admins can insert files" on storage.objects for insert with check ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); create policy "Admins can delete files" on storage.objects for delete using ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); -- 3. Permette ai clienti approvati di leggere solo la propria cartella create policy "Clients can select own files" on storage.objects for select using ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' ); -- 4. Permette ai clienti approvati di caricare i propri documenti create policy "Clients can insert own files" on storage.objects for insert with check ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' ); create policy "Clients can update own files" on storage.objects for update using ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' );

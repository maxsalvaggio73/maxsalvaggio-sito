Guida SQL e Regole di Sicurezza

-- 1. Aggiunge le colonne anagrafiche complete alla tabella profiles alter table public.profiles add column if not exists is_admin boolean default false, add column if not exists file_order jsonb default '[]'::jsonb, add column if not exists full_name text, add column if not exists nationality text, add column if not exists codice_fiscale text, add column if not exists dob date, add column if not exists pob text, add column if not exists birth_province text, add column if not exists birth_country text, add column if not exists address text, add column if not exists residence_city text, add column if not exists residence_province text, add column if not exists residence_country text, add column if not exists phone_number text, add column if not exists social_instagram text, add column if not exists social_facebook text, add column if not exists social_other text, add column if not exists onboarding_completed boolean default false, add column if not exists documents_approved boolean default false; -- 1b. Abilita e configura RLS per la tabella 'profiles' (RISOLVE l'errore RLS) alter table public.profiles enable row level security; create or replace function public.is_admin() returns boolean as $$ select coalesce( (select is_admin from public.profiles where id = auth.uid()), false ); $$ language sql security definer; -- 1c. Policy RLS per profiles create policy "Admins can view all profiles" on public.profiles for select using (public.is_admin()); create policy "Admins can update profiles" on public.profiles for update using (public.is_admin()); create policy "Admins can insert profiles" on public.profiles for insert with check (public.is_admin()); create policy "Users can view own profile" on public.profiles for select using (auth.uid() = id); create policy "Users can update own profile" on public.profiles for update using (auth.uid() = id); -- 2. Permette agli admin la piena gestione del bucket 'client-proofs' create policy "Admins can select files" on storage.objects for select using ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); create policy "Admins can insert files" on storage.objects for insert with check ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); create policy "Admins can delete files" on storage.objects for delete using ( bucket_id = 'client-proofs' and (select is_admin from public.profiles where id = auth.uid()) = true ); -- 3. Permette ai clienti approvati di leggere solo la propria cartella create policy "Clients can select own files" on storage.objects for select using ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' ); -- 4. Permette ai clienti approvati di caricare i propri documenti create policy "Clients can insert own files" on storage.objects for insert with check ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' ); create policy "Clients can update own files" on storage.objects for update using ( bucket_id = 'client-proofs' and (select is_approved from public.profiles where id = auth.uid()) = true and name like (select email from public.profiles where id = auth.uid()) || '/%' ); -- 5. Creazione della tabella per il Portfolio Pubblico create table if not exists public.portfolio_items ( id uuid default gen_random_uuid() primary key, url text not null, storage_path text not null, title text not null, category text not null, project_id text, project_title text, project_place text, project_magazine text, width integer, height integer, is_horizontal boolean, position integer default 0, created_at timestamp with time zone default timezone('utc'::text, now()) not null ); create index if not exists idx_portfolio_items_category_position on public.portfolio_items(category, position); -- 6. Abilita la sicurezza a livello di riga (RLS) per portfolio_items alter table public.portfolio_items enable row level security; -- Politiche RLS per la tabella portfolio_items create policy "Consenti lettura pubblica portfolio_items" on public.portfolio_items for select using (true); create policy "Consenti agli admin la gestione di portfolio_items" on public.portfolio_items for all using ( (select is_admin from public.profiles where id = auth.uid()) = true ) with check ( (select is_admin from public.profiles where id = auth.uid()) = true ); -- 7. Politiche RLS per il bucket 'public-portfolio' (crealo prima nello Storage!) create policy "Consenti lettura pubblica oggetti public-portfolio" on storage.objects for select using (bucket_id = 'public-portfolio'); create policy "Consenti agli admin la gestione di oggetti public-portfolio" on storage.objects for all using ( bucket_id = 'public-portfolio' and (select is_admin from public.profiles where id = auth.uid()) = true ) with check ( bucket_id = 'public-portfolio' and (select is_admin from public.profiles where id = auth.uid()) = true );
